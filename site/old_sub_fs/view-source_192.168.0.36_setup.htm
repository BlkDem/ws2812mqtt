function real_time(hours,min,sec) {
sec=Number(sec)+1;
if (sec>=60){min=Number(min)+1;sec=0;}

if (hours>=24){hours=0};

set_real_time = setTimeout("real_time("+hours+","+min+","+sec+");", 1000);
}
