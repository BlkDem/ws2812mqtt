function set_switch(submit){

else server = "/setswitch?value=0";

function setcbattr()
{
var xhr = new XMLHttpRequest();

// 2. Конфигурируем его: GET-запрос на URL 'phones.json'
xhr.open('GET', 'config.json', false);

// 3. Отсылаем запрос
xhr.send();

// 4. Если код ответа сервера не 200, то это ошибка
if (xhr.status != 200) {
// обработать ошибку
alert( xhr.status + ': ' + xhr.statusText ); // пример вывода: 404: Not Found
} else {
// вывести результат
var myArr = JSON.parse(xhr.responseText);
//document.write('<p>'+ myArr.switch_checked +'</p>');
if (myArr.switch_checked=="")
{
document.getElementById("label1").innerHTML = "<input id='switch_value' type='checkbox' class='l' onclick='set_switch(this);'>";
}
else
{
//alert('checked') ;
document.getElementById("label1").innerHTML = "<input id='switch_value' type='checkbox' checked='1' class='l' onclick='set_switch(this);'>";
}
//document.write('<p>'+ xhr.responseText +'</p>'); // responseText -- текст ответа.
}
}
var timerId = setInterval(function(){setcbattr();}, 3000);
